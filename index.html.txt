getFirestore,
collection,
getDocs,
query,
where,

getDoc,
addDoc
} from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

};

const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

const materiasDiv = document.getElementById("pantallaMaterias");
const alumnosDiv = document.getElementById("pantallaAlumnos");

async function cargarMaterias(){

materiasDiv.innerHTML="<h2>Materias</h2>";

const snapshot = await getDocs(collection(db,"MATERIAS"));

snapshot.forEach(docu=>{

let m = docu.data();

let card = document.createElement("div");
card.className="card";

card.innerHTML = `
<h3>${m.nombre}</h3>
Curso ${m.curso} - Cuatrimestre ${m.cuatrimestre}
`;

card.onclick = ()=>{

verAlumnos(m.materiaId)

};

materiasDiv.appendChild(card);

});

}

async function verAlumnos(materiaId){

materiasDiv.style.display="none";
alumnosDiv.style.display="block";

alumnosDiv.innerHTML=`
<button class="volver" onclick="volver()">Volver</button>
<h2>Lista de alumnos</h2>
`;

const q = query(
collection(db,"inscripciones"),
where("materiaId","==",materiaId)
);

const snapshot = await getDocs(q);

snapshot.forEach(async ins=>{

let alumnoId = ins.data().alumnoId;

const ref = doc(db,"usuarios",alumnoId);
const alumno = await getDoc(ref);

let a = alumno.data();

let div = document.createElement("div");
div.className="alumno";

div.innerHTML=`
<span>${a.nombre}</span>
<span>
<button class="presente" onclick="presente('${alumnoId}','${materiaId}')">P</button>
<button class="ausente" onclick="ausente('${alumnoId}','${materiaId}')">A</button>
</span>
`;

alumnosDiv.appendChild(div);

});

}

window.volver = ()=>{

alumnosDiv.style.display="none";
materiasDiv.style.display="block";

}

window.presente = async (alumnoId,materiaId)=>{

await addDoc(collection(db,"asistencia"),{

alumnoId: alumnoId,
materiaId: materiaId,
fecha: new Date().toISOString(),
presente:true

});

alert("Asistencia guardada");

}

window.ausente = async (alumnoId,materiaId)=>{

await addDoc(collection(db,"asistencia"),{

alumnoId: alumnoId,
materiaId: materiaId,
fecha: new Date().toISOString(),
presente:false

});

alert("Ausente guardado");

}
